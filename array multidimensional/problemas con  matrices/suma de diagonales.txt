#include <iostream>
#include <iomanip>
using namespace std;
int main()
{
    int m[100][100],n;
    cout<<"ingrese el tamaÃ±o de la matriz cuadrada:";
    cin>>n;
    cout<<"introduce los elementos de la matriz:"<<endl;
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<n;j++)
        {
            cout<<"ingrese numero["<<i<<"]["<<j<<"]:";
            cin>>m[i][j];
        }
    }
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<n;j++)
        {
            cout<<m[i][j]<<" ";
        }
        cout<<endl;
    }
    int suma1=0,suma2=0,suma3;
    for(int i=0;i<n;i++)
    {
        suma1+=m[i][i];
        suma2+=m[i][n-i-1];
    }
    suma3=suma1+suma2;
    cout<<"\nmatriz ingresada con la suma en el centro:"<<endl;
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<n;j++)
        {
            if(i==n/2 && j==n/2)
            {
                cout<< setw(4)<<suma3;
            }
            else
            {
                cout<<setw(4)<<m[i][j];
            }
        }
        cout<<endl;
    }
    cout<<"\nsuma de la diagonal principal:"<<suma1<<endl;
    cout<<"suma de la diagonal secundaria:"<<suma2<<endl;
    return 0;
}
